  1. 异步的写法
// 改写
A('file', (err, data) => {
    if (err) {
        console.log(err)
    }
    console.log(data)
    A('file', (err, data2) => {
        if (err) {
            console.log(err)
        }
        console.log(data2)
    })
})

// 完美写法
async function B() {
    try {
        let data = await A('file')
        console.log(data)
        let data2 = await A('file')
        console.log(data)
    } catch (err) {
        console.log(err)
    }
}
  1. 考察Hook
function App() {
    const [count, setCount] = useState(0)
    const [count2, setCount2] = useState(0)
    const fn = useCallback(() => {
        console.log(count)
    }, [count2]) 
    return (
        <div>
            {count}
            <button onClick={() => setCount(100)}>按钮A</button>
            <button onClick={() => setCount2(10000)}>按钮B</button>
            <button onClick={fn}>按钮C</button> 
        </div>
    )
}

// 依次点A, C 输出0
// 依次点A, B, C 输出100
